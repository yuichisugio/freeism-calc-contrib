gh api graphql -F q='is:public archived:false sort:stars-desc' -f query='
  query($q: String!) {
    search(type: REPOSITORY, query: $q, first: 3) {
      nodes {
        ... on Repository {
          nameWithOwner
          url
          stargazerCount
          forkCount
          pullRequests { totalCount }
        }
      }
    }
  }'
