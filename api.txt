gh api graphql -F owner="yoshiko-pg" -F name="difit" -f query='  
    query($name: String!, $owner: String!) {
      repository(owner: $owner, name: $name) {
        pullRequests(first: 100, orderBy: {field: CREATED_AT, direction: DESC}) {
          totalCount
          nodes {
            author {
              login
              ... on User {
                id
                databaseId
              }
            }
          }
        }
      }
      rateLimit{
        cost
        limit
        nodeCount
        used
        remaining
        resetAt
      }
    }
  '
